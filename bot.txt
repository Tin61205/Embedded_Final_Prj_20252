//--------------------------------------------------------------
// BFS helper: kiểm tra có thể đi từ (cx,cy) sang (nx,ny) theo dir không.
// Dùng kiểm tra 2 chiều: ô hiện tại có cửa ra HOẶC ô kế có cửa vào.
// Cách này cho phép đi xuyên qua gate cells (có BGATE_D nhưng không có ROOM_DOOR_D).
//--------------------------------------------------------------
static uint32_t bot_bfs_can_traverse(uint32_t cx, uint32_t cy,
                                     uint32_t nx, uint32_t ny,
                                     uint32_t dir) {
    uint32_t out_door, in_door;
    switch (dir) {
        case MOVE_UP:    out_door = ROOM_DOOR_U; in_door = ROOM_DOOR_D; break;
        case MOVE_RIGHT: out_door = ROOM_DOOR_R; in_door = ROOM_DOOR_L; break;
        case MOVE_DOWN:  out_door = ROOM_DOOR_D; in_door = ROOM_DOOR_U; break;
        case MOVE_LEFT:  out_door = ROOM_DOOR_L; in_door = ROOM_DOOR_R; break;
        default: return 0;
    }
    // Cho phép đi nếu: ô hiện tại có cửa ra HOẶC ô kế có cửa vào (xuyên gate)
    if (((Maze.Room[cx][cy].door & out_door) == 0) &&
        ((Maze.Room[nx][ny].door & in_door) == 0)) {
        return 0;
    }
    return bot_is_walkable(nx, ny, 1);
}
//--------------------------------------------------------------
// BFS tìm hướng đầu tiên tối ưu từ (x1,y1) đến (x2,y2).
// Tránh portal (bot_avoid_portal phải được set = 1 trước khi gọi).
// Trả về MOVE_STOP nếu không tìm thấy đường.
//--------------------------------------------------------------
static uint32_t bot_calc_move_bfs_home(uint32_t x1, uint32_t y1,
                                       uint32_t x2, uint32_t y2) {
    // Static arrays để tránh stack overflow trên embedded (tổng ~3.5KB)
    static uint8_t visited[ROOM_CNT_X][ROOM_CNT_Y];
    static uint8_t first_move_arr[ROOM_CNT_X][ROOM_CNT_Y];
    static uint8_t q_x[ROOM_CNT_X * ROOM_CNT_Y];
    static uint8_t q_y[ROOM_CNT_X * ROOM_CNT_Y];
    uint32_t q_head, q_tail;
    uint32_t cx, cy, nx, ny, fm;
    if (x1 == x2 && y1 == y2) return MOVE_STOP;
    if (x1 >= ROOM_CNT_X || y1 >= ROOM_CNT_Y) return MOVE_STOP;
    if (x2 >= ROOM_CNT_X || y2 >= ROOM_CNT_Y) return MOVE_STOP;
    memset(visited, 0, sizeof(visited));
    q_head = 0;
    q_tail = 0;
    visited[x1][y1] = 1;
    first_move_arr[x1][y1] = (uint8_t)MOVE_STOP;
    q_x[q_tail] = (uint8_t)x1;
    q_y[q_tail] = (uint8_t)y1;
    q_tail++;
    while (q_head < q_tail) {
        cx = q_x[q_head];
        cy = q_y[q_head];
        fm = first_move_arr[cx][cy];
        q_head++;
        if (cx == x2 && cy == y2) {
            return fm; // Tìm thấy đích, trả về hướng đầu tiên
        }
        // Mở rộng 4 hướng
        // UP
        if (cy > 0) {
            nx = cx; ny = cy - 1;
            if (!visited[nx][ny] && bot_bfs_can_traverse(cx, cy, nx, ny, MOVE_UP)) {
                visited[nx][ny] = 1;
                first_move_arr[nx][ny] = (fm == MOVE_STOP) ? (uint8_t)MOVE_UP : (uint8_t)fm;
                q_x[q_tail] = (uint8_t)nx;
                q_y[q_tail] = (uint8_t)ny;
                q_tail++;
            }
        }
        // RIGHT
        if (cx < ROOM_CNT_X - 1) {
            nx = cx + 1; ny = cy;
            if (!visited[nx][ny] && bot_bfs_can_traverse(cx, cy, nx, ny, MOVE_RIGHT)) {
                visited[nx][ny] = 1;
                first_move_arr[nx][ny] = (fm == MOVE_STOP) ? (uint8_t)MOVE_RIGHT : (uint8_t)fm;
                q_x[q_tail] = (uint8_t)nx;
                q_y[q_tail] = (uint8_t)ny;
                q_tail++;
            }
        }
        // DOWN
        if (cy < ROOM_CNT_Y - 1) {
            nx = cx; ny = cy + 1;
            if (!visited[nx][ny] && bot_bfs_can_traverse(cx, cy, nx, ny, MOVE_DOWN)) {
                visited[nx][ny] = 1;
                first_move_arr[nx][ny] = (fm == MOVE_STOP) ? (uint8_t)MOVE_DOWN : (uint8_t)fm;
                q_x[q_tail] = (uint8_t)nx;
                q_y[q_tail] = (uint8_t)ny;
                q_tail++;
            }
        }
        // LEFT
        if (cx > 0) {
            nx = cx - 1; ny = cy;
            if (!visited[nx][ny] && bot_bfs_can_traverse(cx, cy, nx, ny, MOVE_LEFT)) {
                visited[nx][ny] = 1;
                first_move_arr[nx][ny] = (fm == MOVE_STOP) ? (uint8_t)MOVE_LEFT : (uint8_t)fm;
                q_x[q_tail] = (uint8_t)nx;
                q_y[q_tail] = (uint8_t)ny;
                q_tail++;
            }
        }
    }
    return MOVE_STOP; // Không tìm được đường
}
//--------------------------------------------------------------
// bot strategy : home
//                 move to the room with the shortest distance
//                 to the home position
//                 (but don't go backwards to avoid toggling)
//   Dùng BFS để tìm đường ngắn nhất thực sự về điểm hồi sinh.
//   BFS đảm bảo ghost không bao giờ bị kẹt trong dead-end hoặc
//   hốc chữ M vì nó tìm đường TỐI ƯU toàn bộ mê cung.
//   Fallback về greedy nếu BFS không tìm thấy (không nên xảy ra).
//--------------------------------------------------------------
uint32_t bot_calc_move_home(uint32_t ghost, uint32_t xp, uint32_t yp, uint32_t akt_dir) {
    uint32_t ret_wert = MOVE_STOP;
        typ = CLYDE_HOME_Y;
    }
    // Bật cờ tránh portal khi ghost Dead tìm đường về nhà
    // BFS: tránh portal, tìm đường tối ưu thực sự qua mê cung
    bot_avoid_portal = 1;
    ret_wert = bot_calc_move(xp, yp, txp, typ, akt_dir);
    ret_wert = bot_calc_move_bfs_home(xp, yp, txp, typ);
    bot_avoid_portal = 0;
    return (ret_wert);
    // Fallback về greedy nếu BFS thất bại (bảo vệ an toàn)
    if (ret_wert == MOVE_STOP) {
        bot_avoid_portal = 1;
        ret_wert = bot_calc_move(xp, yp, txp, typ, akt_dir);
        bot_avoid_portal = 0;
    }
    return ret_wert;
}
